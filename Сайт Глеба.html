        function sendToTelegram() {

            const name = document.getElementById('name').value;

            const email = document.getElementById('email').value;

            const message = document.getElementById('message').value;

            

            if (name && email && message) {

                const botToken = "YOUR_BOT_TOKEN"; // ЗАМЕНИТЬ НА ТОКЕН БОТА

                const chatId = "YOUR_CHAT_ID"; // ЗАМЕНИТЬ НА ID ЧАТА

                const text = `Новая заявка!\nИмя: ${name}\nEmail: ${email}\nСообщение: ${message}`;

                

                fetch(`https://api.telegram.org/bot${botToken}/sendMessage?chat_id=${chatId}&text=${encodeURIComponent(text)}`)

                    .then(response => response.json())

                    .then(data => alert("Заявка отправлена!"))

                    .catch(error => alert("Ошибка отправки!"));

            } else {

                alert("Заполните все поля!");

            }

        }

    </script>
